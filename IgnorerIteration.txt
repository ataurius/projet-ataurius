package com.projetataurius;

public class IgnorerItiration {

	public static void main(String[] args) {
		fonctionContinue();
	}

	public static void fonctionContinue() {
		for (int i = 0; i < 10; i++) {
			// déclarations exécutées à chaque itération
			if (i == 2 || i == 5) {
				continue;
			}
			System.out.println("Valeur de i : " + i + ".");
		}

	}
}
