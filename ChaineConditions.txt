package com.projetataurius;

public class ChaineConditions {

	public static void main(String[] args) {
		fonction1();
	}

	public static void fonction1() {

		int nombre = 0;

		if (nombre == 1) {
			console("option 1");
		} else if (nombre == 2) {
			console("option 2");
		} else {
			console("autres options");
		}

		switch (nombre) {
		case 1:
			console("modèle 1");
			break;
		default:
			console("autres modèles");
		}
	}

	static void console(String texte) {
		System.out.println(texte);
	}
}
